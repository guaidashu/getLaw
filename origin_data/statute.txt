	function validate(keyword) {
		var pattern = new RegExp(
				"[%--`~!@#$^&*()=|{}':;',\\[\\]\\\\.<>/?~！@#￥……&*（）——|{}【】‘；：”“'。，、？]"); //格式 RegExp("[在中间定义特殊过滤字符]")  

		var rs = "";
		for ( var i = 0; i < keyword.length; i++) {
			rs = rs + keyword.substr(i, 1).replace(pattern, '');
		}
		return rs;
	}
